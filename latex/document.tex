	% --- 1. INTRODUCCIÓN ---
	\section{Introducción a Rust}
	\begin{frame}{¿Qué es Rust?}
		\begin{block}{Definición}
			Rust es un lenguaje de programación de sistemas compilado, diseñado para ser seguro, concurrente y práctico. Fue patrocinado originalmente por Mozilla Research en 2010 y actualmente es impulsado por la \textit{Rust Foundation}.
		\end{block}
		
		\vspace{0.5cm}
		
		\textbf{Propósito principal:}
		Resolver problemas comunes en lenguajes como C y C++, especialmente los relacionados con la gestión de memoria, sin sacrificar el rendimiento.
	\end{frame}
	
	% --- 2. PARADIGMAS ---
	\section{Paradigmas Soportados}
	\begin{frame}{Paradigmas Soportados}
		Rust soporta activamente:
		
		\begin{itemize}
			\item \textbf{Imperativo/Procedimental}.
			\item \textbf{Funcional}.
			\item \textbf{Orientado a Objetos (Limitado):} Permite encapsulamiento e interfaces mediante \textit{traits}, pero \textbf{no admite herencia tradicional\footnote{Rust prefiere la composición sobre la herencia.}} de clases.
		\end{itemize}
	\end{frame}

	% --- 3. CARACTERÍSTICAS CLAVE ---
	\section{Características Principales}
	
	\begin{frame}{El Sistema de Ownership}
		La innovación más grande de Rust es cómo maneja la memoria \textbf{sin usar un Garbage Collector}.
		
		\vspace{0.3cm}
		
		\begin{block}{Ownership}
			\begin{enumerate}
				\item Cada valor en Rust tiene una variable que es su \textit{propietario} (owner).
				\item Solo puede haber un propietario a la vez.
				\item Cuando el propietario sale del scope, el valor se elimina de la memoria automáticamente.
			\end{enumerate}
		\end{block}
	\end{frame}
	
	% --- 4. ECOSISTEMA ---
	\section{Ecosistema y Herramientas}
	\begin{frame}{Ecosistema y Herramientas}
		Rust viene con las siguientes herramientas:
		
		\begin{itemize}
			\item \textbf{Cargo:} Es el gestor de paquetes y sistema de compilación de Rust. Permite compilar código, correr pruebas y generar documentación.
			\item \textbf{Crates.io:} Es el registro oficial de paquetes (llamados \textit{crates}).
			\item \textbf{Rustfmt y Clippy:} Herramientas nativas para formatear el código y analizar errores comunes o malas prácticas.
		\end{itemize}
	\end{frame}
	
	
	
	
	
	% --- 5. CONCLUSIÓN ---
	\section{Conclusión}
	\begin{frame}{¿Por qué elegir Rust?}
		\begin{columns}
			\column{0.5\textwidth}
			\textbf{Ventajas}
			\begin{itemize}
				\item Rendimiento muy alto (similar a C++).
				\item Seguridad de memoria garantizada en tiempo de compilación.
				\item Concurrencia.
			\end{itemize}
			
			\column{0.5\textwidth}
			\textbf{Desafíos}
			\begin{itemize}
				\item Curva de aprendizaje elevada.
				\item Tiempos de compilación más lentos.
				\item El compilador es muy estricto, pero te obligará a escribir código seguro.
			\end{itemize}
		\end{columns}
	\end{frame}
